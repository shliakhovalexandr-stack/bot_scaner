    <h2>Наведите на акцизную марку</h2>
    <div id="reader"></div>
    <script>
        const tg = window.Telegram.WebApp;
        tg.expand();

        const html5QrCode = new Html5Qrcode("reader");
        
        // Настройки сканера (ищем штрих-коды и DataMatrix)
        const config = { fps: 15, qrbox: { width: 250, height: 150 } };

        html5QrCode.start({ facingMode: "environment" }, config, (decodedText) => {
            // Как только камера «поймала» код:
            tg.sendData(decodedText); // Бот получит этот текст
            tg.close(); // Окно закроется само
        }).catch(err => {
            console.error("Ошибка камеры:", err);
        });
    </script>
